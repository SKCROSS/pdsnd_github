#define final column names, including new city variable

# one, plot frequency graphs to see distribution split between different cities, converting seconds to minutes and zooming in since many extreme values  
ggplot(data = subset(bshare, !is.na(Trip.Duration)), mapping = aes(x = Trip.Duration/60, color = City)) +
  geom_freqpoly(binwidth = 1) + 
  coord_cartesian(xlim = c(0, 180)) + 
  labs(title="Frequency polygons examining the distribution for trip duration, split by main user type", 
       x="Trip duration in minutes", y="Frequency")

#insight: NY has more of a high peak than the other cities, but all 3 have the same right-skewed distribution meaning most trips have shorter durations but there is a long right tail and many outliers.

# two, compute numerical summary statistic for average trip duration split by city 
by(bshare$Trip.Duration/60, bshare$City, summary)

# insight: similar median values across all three cities (10-12) but a higher mean (21) for Washington than NY (15) or Chicago (16)

# three, assume (for the sake of showcasing a loop) that the outliers/extreme values for trip duration need treating. 
# to minimise the effects of extreme observations I will set all outliers to a specified percentile of the data

# four, set caps to the 1st and 99th percentiles  
caps <- quantile(subset(bshare, !is.na(Trip.Duration))$Trip.Duration, probs=c(.01, .91), na.rm = T)

# five, use a loop to cap extreme values 
idx = 1

for (num in subset(bshare, !is.na(Trip.Duration))$Trip.Duration){
  if (num < caps[1]){
    bshare$Trip.Duration[idx] = caps[1]
  } 
  else if(num > caps[2]) {
    bshare$Trip.Duration[idx] = caps[2]
  }
  idx = idx + 1
}

# six, recompute numerical summary statistics for average trip duration split by city  
by(bshare$Trip.Duration/60, bshare$City, summary)

# insight: median values remain as was, but mean values decrease - mostly for Washington (18), but also for NY (14) and Chicago (15).

  geom_bar(mapping = aes(x=Month), color = 'black', fill = '#F79420') + 
  facet_wrap(bshare$City) + 
  ggtitle('Bar charts of trips made, split by month and city') +
  labs(x = 'Month, by city', y = 'Number of trips')

# days
